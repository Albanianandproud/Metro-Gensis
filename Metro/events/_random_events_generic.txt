add_namespace = random_event

## Vagrants in the Tunnel ##
country_event = {
	id = random_event.1
	title = random_event.1.t
	desc = random_event.1.d
	picture = event_tunnel

	mean_time_to_happen = {
		days = 3600
	}

	trigger = {
		NOT = {
			original_tag = GWC
		}
		has_war = no
		num_of_controlled_states > 1
	}

	option = {	## Let the Join ##
		name = random_event.1.a
		custom_effect_tooltip = random_event.1.a_tt
		random_owned_state = {
				limit = {
					NOT = {
					    has_state_category = pastoral
					}
				}
			add_manpower = 20
		}
		add_stability = -0.05
	}

	option = {	## Turn them Away ##
		custom_effect_tooltip = random_event.1.b_tt
		name = random_event.1.b
	}
}

## Found Valuables ##
country_event = {
	id = random_event.5
	title = random_event.5.t
	desc = random_event.5.d
	picture = event_station_defense_2
	is_triggered_only = yes

	option = {	## Sweet ##
		name = random_event.5.a

		random_list = {
		    33 = { ## Found Resources 
		        capital_scope = {
			    	add_resource = {
			        	type = alloys
			        	amount = 2
			    	}
			    	add_resource = {
			        	type = scrap
			        	amount = 2
			    	}
				}
		    }
		    24 = { ## Found Arms
		    	add_equipment_to_stockpile = {
		        	type = infantry_equipment
		        	amount = 20
		    	}      
		    }
		    33 = { ## Found Valuable Trinkets
		        add_political_power = 50
		    }
		}
	}
}
