STB_supersoldaten_projekt_scripted_effect = {

	## custom_effect_tooltip ## 
	every_unit_leader = {
		if = {
			limit = {
				has_trait = fainthearted_personality_trait
			}
			remove_unit_leader_trait = fainthearted_personality_trait
		}
		if = {
			limit = {
				has_trait = paranoid_personality_trait
			}
			remove_unit_leader_trait = paranoid_personality_trait
		}
		if = {
			limit = {
				has_trait = irrational_personality_trait
			}
			remove_unit_leader_trait = irrational_personality_trait
		}
		if = {
			limit = {
				has_trait = spineless_personality_trait
			}
			remove_unit_leader_trait = spineless_personality_trait
		}

		if = {
			limit = {
				NOT = {
					has_trait = supersoldaten_personality_trait
				}
			}
			add_trait = {
				trait = supersoldaten_personality_trait
			}

			add_skill_level = 1
		}

		if = {
			limit = {
				has_trait = supersoldaten_personality_trait
			}
			hidden_effect = { remove_unit_leader_trait = supersoldaten_personality_trait }
			add_unit_leader_trait = supersoldaten_personality_trait_2

			add_skill_level = 1
		}
	}
}
