SAV_set_disintegrated_cosmetic_tag = {
	if = {
		limit = { original_tag = SAV }
		if = {
			limit = { has_country_flag = SAV_disintegrated_flag }
			if = {
				limit = { has_government = council }
				set_cosmetic_tag = SAV_disintegrated_council
			}
			else_if = {
				limit = { has_government = oligarch }
				set_cosmetic_tag = SAV_disintegrated_oligarch
			}
			else_if = {
				limit = { has_government = dictatorial }
				set_cosmetic_tag = SAV_disintegrated_dictatorial
			}
			else = {
				set_cosmetic_tag = SAV_disintegrated_community
			}
		}
	}
}
