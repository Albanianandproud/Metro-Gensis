reclaim_metro = {

 	reclaim_this_state = {
        icon = decision_generic_military
		cost = 20 ### Change Cost depending on type of abandoned state? 
		days_remove = 50
		days_re_enable = 5
		on_map_mode = map_only
		state_target = yes
		fixed_random_seed = no

        ai_will_do = {
			factor = 5
			modifier = {
				factor = -200
				has_war = yes
			}
			modifier = {
				is_major = yes
				factor = -200
			}
		}
		
        visible = {
            RUI = {
                owns_state = FROM
            }
            FROM = {
                any_neighbor_state = {
                    is_controlled_by = ROOT
					NOT = {
						state = 1
						state = 237
					}
                }
            }
        }

		available = {
			FROM = {
                is_owned_by = RUI
			}
			has_manpower > 10
			has_equipment = {
                infantry_equipment > 9
                supplies_equipment > 1
            }
			NOT = {
				ROOT = { has_country_flag = reclaimation_underway }
				FROM = {
					OR = {
						has_state_flag = state_being_colonised
						state = 1
						state = 237
					}
				}
			}
		}

		complete_effect = {
			add_manpower = -10
		    add_equipment_to_stockpile = {
		        type = infantry_equipment
		        amount = -10
		    }
		    add_equipment_to_stockpile = {
		        type = supplies_equipment
		        amount = -2
		    }
			set_country_flag = reclaimation_underway
			FROM = {
				set_state_flag = state_being_colonised
			}
		}

		remove_effect = {
		    random_list = {
		        50 = {
					ROOT = {
						set_variable = {
							var = state_colonised_id
							value = FROM
						}
					}
		            ROOT = { country_event = reclaimation_event.2 }
		            FROM = {
						clr_state_flag = state_being_colonised
					}
					modifier = {
						factor = 0.25
						FROM = {
							OR = {
								has_state_flag = flooded_tunnel_flag
								has_state_flag = anomaly_tunnel_flag
							}
						}
					}
		        }
		        25 = { #Overwhelmed Mutants
		            country_event = reclaimation_event.mutants
		            FROM = {
						clr_state_flag = state_being_colonised
					}
		        }
		        25 = { #Expedition lost
		            country_event = reclaimation_event.lost
		            FROM = {
						clr_state_flag = state_being_colonised
					}
		        }
			}
			ROOT = { clr_country_flag = reclaimation_underway }
		}
		ai_will_do = {
			factor = 0.5
			modifier = {
				factor = 1.5
				is_major = no
			}
		}
    }
}
