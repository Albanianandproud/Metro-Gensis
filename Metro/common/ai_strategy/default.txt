build_intelligence_agency = {
	enable = {
		date > 1940.1.1
		has_war = no
		num_of_civilian_factories > 25
	}
	abort = {

			has_war = yes
			num_of_civilian_factories < 25
		}

	ai_strategy = {
		type = intelligence_agency_usable_factories
		value = 5
	}

}
