	scavenge_scrap = {

		icon = scrap

		available = {
		    has_manpower > 9
			NOT = { has_country_flag = scavenging_scrap }
		}

		visible = {
			always = yes
		}

		fire_only_once = no

		complete_effect = {
		    add_manpower = -10
			set_country_flag = scavenging_scrap
		}

		days_remove = 50
		days_re_enable = 0

		cost = 30

		remove_effect = {
		    add_manpower = 5
			random_owned_controlled_state = {
		        limit = {
			        NOT = {
			            has_state_category = pastoral
				    }
		        }
			    add_resource = {
			        type = scrap
			        amount = 2
			    }
		    }

			clr_country_flag = scavenging_scrap
		}
		ai_will_do = {
			factor = 15
			modifier = {
				factor = 15
				scrap < 5
			}
			modifier = {
				factor = 10
				scrap < 10
			}
			modifier = {
				factor = 0.1
				has_manpower < 30
			}
		}
    }
