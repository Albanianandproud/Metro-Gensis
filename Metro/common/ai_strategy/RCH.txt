RCH_kill_SAV = {
	enable = {
		focus_progress = {
			focus = RCH_expansion
			progress > 0.01
		}
		NOT = {
			has_completed_focus = RCH_expansion
		}
		country_exists = SAV
	}

	abort = {
		OR = {
			NOT = { country_exists = SAV }

			AND = {
				RCH = { has_war_with = SAV }
			}
		}
	}

	ai_strategy = {
		type = conquer
		id = "SAV"
		value = 100
	}

	ai_strategy = {
    	type = build_building
    	id = infrastructure
    	target = 426 
    	value = 3
	}
}
