## Vagrants in the Tunnel ##
country_event = {
	id = random_event.1
	title = random_event.1.t
	desc = random_event.1.d
	picture = "GFX_event (23)"

	mean_time_to_happen = {
		days = 3650
	}

	trigger = {
		NOT = {
			original_tag = GWC
			original_tag = RED
			original_tag = MSU
		}
		has_war = no
		num_of_controlled_states > 2
	}

	option = {	## Let the Join ##
		name = random_event.1.a
		custom_effect_tooltip = random_event.1.a_tt
		random_owned_state = {
				limit = {
					NOT = {
					    has_state_category = pastoral
					}
				}
			add_manpower = 25
		}
		add_stability = -0.04
	}

	option = {	## Turn them Away ##
		custom_effect_tooltip = random_event.1.b_tt
		name = random_event.1.b
	}
}

	}
