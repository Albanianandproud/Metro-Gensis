EUR_decrease_mil_power = {
	if = {
		limit = { has_idea = EUR_mil_power_1 }
		remove_ideas = EUR_mil_power_1
		set_country_flag = eur_mil_reorganised
	}
	else_if = {
		limit = { has_idea = EUR_mil_power_2 }
		swap_ideas = {
			remove_idea = EUR_mil_power_2
			add_idea = EUR_mil_power_1
		}
	}
	else_if = {
		limit = { has_idea = EUR_mil_power_3 }
		swap_ideas = {
			remove_idea = EUR_mil_power_3
			add_idea = EUR_mil_power_2
		}
	}
	else_if = {
		limit = { has_idea = EUR_mil_power_4 }
		swap_ideas = {
			remove_idea = EUR_mil_power_4
			add_idea = EUR_mil_power_3
		}		
	}
	else_if = {
		limit = { has_idea = EUR_mil_power_5 }
		swap_ideas = {
			remove_idea = EUR_mil_power_5
			add_idea = EUR_mil_power_4
		}
	}
}
