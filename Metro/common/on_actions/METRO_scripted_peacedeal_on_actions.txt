on_actions = {
	#ROOT is capitulated country, FROM is winner.

	## PAG / TSH
	on_capitulation_immediate = { 
		effect = {
			if = { 
				limit = {
					FROM = { original_tag = PAG }
					ROOT = { original_tag = TSH }
				}
				FROM = { annex_country = { target = PREV } }
			} 
		}
	}

	## MUT / SPT
	on_capitulation_immediate = { 
		effect = {
			if = { 
				limit = {
					FROM = { original_tag = SPT }
					ROOT = { original_tag = MUT }
				}
				FROM = { annex_country = { target = PREV } }
			} 
		}
	}
}
