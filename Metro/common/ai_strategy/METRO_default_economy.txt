METRO_division_limiter = {
	enable = {
		date < 2035.1.1
		check_variable = { bullet < 649 }
		check_variable = { total_balance_temp < 0 }
		NOT = { surrender_progress > 0.24 }
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = build_army
		id = motorized
		value = -3000
	}
	ai_strategy = {
		type = build_army
		id = armor
		value = -3000
	}
	ai_strategy = {
		type = build_army
		id = infantry
		value = -3000
	}
	ai_strategy = {
		type = build_army
		id = marines
		value = -3000
	}
	ai_strategy = { type = force_build_armies value = 0 }
}

METRO_reduce_infantry_equipment_at_high_IC = {
	enable = {
		OR = {
			num_of_military_factories > 14
			AND = {
				num_of_military_factories > 20
				stockpile_ratio = {
					archetype = infantry_equipment
					ratio > 1
				}
			}
		}
	}
	abort = { always = no }
	ai_strategy = {
		type = equipment_production_factor
		id = infantry
		value = -15
	}
}

METRO_dont_forget_to_produce_trains = {
