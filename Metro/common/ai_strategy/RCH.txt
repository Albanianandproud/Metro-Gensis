RCH_kill_DIM = {
	enable = {
		focus_progress = {
			focus = RCH_expansion
			progress > 0.01
		}
		NOT = {
			has_completed_focus = RCH_expansion
		}
		country_exists = DIM
	}

	abort = {
		OR = {
			NOT = { country_exists = DIM }

			AND = {
				RCH = { has_war_with = DIM}
			}
		}
	}

	ai_strategy = {
		type = conquer
		id = "DIM"
		value = 100
	}
}
